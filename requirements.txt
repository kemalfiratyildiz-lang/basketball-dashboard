import streamlit as st
import pandas as pd
import plotly.graph_objects as go

# SAYFA AYARI
st.set_page_config(
    page_title="Asist / Top Kaybı Dashboard",
    layout="wide"
)

# ARKA PLAN
st.markdown("""
<style>

.stApp {
    background: linear-gradient(
        135deg,
        #f8fafc 0%,
        #dbeafe 50%,
        #f1f5f9 100%
    );
}

.block-container {
    padding-top: 2rem;
}

h1, h2, h3, p {
    color: #0f172a !important;
}

</style>
""", unsafe_allow_html=True)

# BAŞLIK
st.title("🏀 Asist / Top Kaybı Dashboard")

st.markdown("""
10+ dakika oynayan ve en az 7 maça çıkan oyuncuların  
interaktif asist / top kaybı analizi.
""")

# CSV OKU
df = pd.read_csv("bsl_stats.csv")

# KOLONLAR
player_col = "Player"
team_col = "Team"
minute_col = "MPG"
games_col = "GP"
assist_col = "APG"
turnover_col = "TOV"

# SAYISAL ÇEVİR
for col in [minute_col, games_col, assist_col, turnover_col]:
    df[col] = pd.to_numeric(df[col], errors="coerce")

# TEMİZLE
df = df.dropna(subset=[
    player_col,
    team_col,
    minute_col,
    games_col,
    assist_col,
    turnover_col
])

# FİLTRE
df = df[
    (df[minute_col] >= 10) &
    (df[games_col] >= 7) &
    (df[turnover_col] > 0)
].copy()

# AST/TOV
df["AST_TOV_RATIO"] = (
    df[assist_col] / df[turnover_col]
)

# GRAFİK
fig = go.Figure()

fig.add_trace(go.Scatter(

    x=df[assist_col],
    y=df[turnover_col],

    mode="markers+text",

    text=df[player_col],

    textposition="top center",

    textfont=dict(
        size=11,
        color="#0f172a",
        family="Arial Black"
    ),

    marker=dict(

        size=df[minute_col] * 1.2,

        color=df["AST_TOV_RATIO"],

        colorscale="Viridis",

        showscale=True,

        opacity=0.92,

        line=dict(
            width=1.5,
            color="white"
        ),

        colorbar=dict(
            title="AST/TOV"
        )
    ),

    customdata=df[[
        team_col,
        games_col,
        minute_col,
        "AST_TOV_RATIO"
    ]],

    hovertemplate=
        "<b>%{text}</b><br>" +
        "Takım: %{customdata[0]}<br>" +
        "Maç: %{customdata[1]}<br>" +
        "Dakika: %{customdata[2]:.1f}<br>" +
        "Asist: %{x:.1f}<br>" +
        "Top Kaybı: %{y:.1f}<br>" +
        "AST/TOV: %{customdata[3]:.2f}<extra></extra>"
))

# TASARIM
fig.update_layout(

    title=dict(
        text="10+ Dakika & 7+ Maç: Asist vs Top Kaybı",
        x=0.5,

        font=dict(
            size=30,
            color="#0f172a",
            family="Arial Black"
        )
    ),

    height=900,

    plot_bgcolor="#ffffff",

    paper_bgcolor="rgba(255,255,255,0)",

    font=dict(
        color="#0f172a",
        family="Arial"
    ),

    xaxis=dict(

        title="Asist Ortalaması (APG)",

        gridcolor="rgba(0,0,0,0.12)",

        zeroline=False,

        showline=True,

        linewidth=2,

        linecolor="black",

        tickfont=dict(
            size=18,
            color="black",
            family="Arial Black"
        ),

        title_font=dict(
            size=24,
            color="black",
            family="Arial Black"
        )
    ),

    yaxis=dict(

        title="Top Kaybı Ortalaması (TOV)",

        gridcolor="rgba(0,0,0,0.12)",

        zeroline=False,

        showline=True,

        linewidth=2,

        linecolor="black",

        tickfont=dict(
            size=18,
            color="black",
            family="Arial Black"
        ),

        title_font=dict(
            size=24,
            color="black",
            family="Arial Black"
        )
    ),

    margin=dict(
        l=140,
        r=90,
        t=100,
        b=90
    )
)

# GRAFİĞİ GÖSTER
st.plotly_chart(
    fig,
    use_container_width=True
)

# TABLO
st.subheader("Oyuncu Tablosu")

st.dataframe(

    df[[
        player_col,
        team_col,
        games_col,
        minute_col,
        assist_col,
        turnover_col,
        "AST_TOV_RATIO"
    ]].sort_values(
        "AST_TOV_RATIO",
        ascending=False
    ),

    use_container_width=True
)
